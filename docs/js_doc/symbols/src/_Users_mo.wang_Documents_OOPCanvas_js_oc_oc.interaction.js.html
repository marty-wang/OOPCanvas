  9     // ## init ##

 11     OC.initialize( function (oc) {
 12         oc._interaction = new Interaction(oc);
 13 
 14         debug.info("interaction module is init'ed.");
 15     });
 16 
 17     // ++ Augment OOPCanvas ++
 18     // These are protected methods and meant to be used internally across
 19     // 1st-party modules.
 20 
 21     var fn = OC.prototype;
 22 
 23     fn._dequeueEvent = function () {
 24         return this._interaction.dequeueEvent();
 25     };
 26 
 27     fn._hitTest = function(element, x, y) {
 28         return this._interaction.hitTest(element, x, y);
 29     };
 30 
 31     fn._getHitTestContext = function() {
 32         return this._interaction.getHitTestContext();
 33     };
 34     
 35     // == Interaction ==
 36     
 37 
 38     /**
 39      * @name Interaction
 40      * @class This class handles the interaction with users. <br/><br/>
 41      * <i>This class is <b>NOT</b> exposed for public use, as it functions as 
 42      * a singleton per OOPCanvas instance, and is managed by its OOPCanvas 
 43      * instance. However, it adds functions to OOPCanvas, which in turn interacts with
 44      * other 1st-party modules.</i>
 45      */
 46 
 47     function Interaction (oc) {
 48         this._oc = oc;
 49         this._canvas = oc.getCanvas();
 50         this._refPoint = oc.getPosition();
 51 
 52         this._hitTestCtx = null;
 53 
 54         this._events = []; // queue-ish, first-in-first-out
 55 
 56         _createHitTestContext(this);        
 57         _registerEventHandlers(this);
 58     }
 59 
 60     Interaction.prototype.getHitTestContext = function() {
 61         return this._hitTestCtx;
 62     };
 63 
 64     Interaction.prototype.hitTest = function(element, x, y) {
 65         var ctx = this._hitTestCtx;
 66         var result = null;
 67         if ( ctx.isPointInPath(x, y) ) {
 68             result = {
 69                 'element': element,
 70                 'point': [x, y]
 71             };
 72         }
 73         return result;
 74     };
 75 
 76     Interaction.prototype.dequeueEvent = function() {
 77         return _dequeueEvent(this);
 78     };
 79 
 80     function _createHitTestContext (ia) {
 81         var oc = ia._oc;
 82         var buffer = document.createElement('canvas');
 83         var size = oc.getSize();
 84         buffer.width = size.width;
 85         buffer.height = size.height;
 86         ia._hitTestCtx = buffer.getContext('2d');
 87     }
 88 
 89     var events = ["click", "mousemove", "mousedown", "mouseup"];
 90 
 91     function _registerEventHandlers (ia) {
 92         var canvas = ia._canvas;
 93 
 94         // Be careful with mouseover and mouseout of canvas
 95         // because they may or may not intervene with
 96         // the same-named event relay to UIElement.
 97         
 98         for ( var i = 0, count = events.length ; i < count; i++ ) {
 99             _bind(canvas, events[i], function(evt){
100                 //debug.debug("interaction: " + evt.type);
101                 _enqueueEvent(ia, evt);
102             });
103         }
104     }
105 
106     // == Event handlers ==
107 
108     function _onmousemove (ia, evt) {
109         debug.debug("interaction: on mouse move");
110         _enqueueEvent(ia, evt);
111     }
112 
113     function _onclick (ia, evt) {
114         debug.debug("interaction: on click");
115         _enqueueEvent(ia, evt);
116     }
117 
118     // == Private ==
119 
120     // TODO: normalize event handling
121     function _bind (canvas, eventName, handler) {
122         canvas.addEventListener(eventName, handler, false);
123     }
124     
125     // TODO: normalize event 
126     function _enqueueEvent (ia, event) {
127         event = !event ? window.event : event;
128         var relPoint = _getRelativePoint(event, ia._refPoint);
129         var evt = {
130             'type': event.type,
131             'left': relPoint[0],
132             'top': relPoint[1]
133         };
134         ia._events.push(evt);
135     }
136 
137     function _dequeueEvent (ia) {
138         return ia._events.shift();
139     }
140 
141     function _getRelativePoint(e, refPoint) {
142 	    var posx = 0;
143 	    var posy = 0;
144 	    if (e.pageX || e.pageY) {
145 		    posx = e.pageX;
146 		    posy = e.pageY;
147 	    } else if (e.clientX || e.clientY) {
148 		    posx = e.clientX + document.body.scrollLeft;
149 		    posy = e.clientY + document.body.scrollTop;
150 	    }
151 	    return [ posx - refPoint.left, posy - refPoint.top ];
152     }
153 
154     debug.info("interaction module is installed.");
155 
156 })(OOPCanvas);
