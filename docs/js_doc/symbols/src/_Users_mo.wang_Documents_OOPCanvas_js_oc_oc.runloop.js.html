 77     // ===================
 78     // == Runloop Class ==
 79     // ===================
 80 
 81     /**
 82      * @name Runloop 
 83      * @class Runloop allows other objects to install hooks on it, and
 84      * thereafter sends them ticks at the fps as configured.<br/>
 85      * This class is <b>NOT</b> exposed for public use, as it functions as 
 86      * a singleton per OOPCanvas instance, and is managed by its OOPCanvas 
 87      * instance. However, it adds functions to OOPCanvas, which in turn interacts with
 88      * other modules and public objects.
 89      * @description Here is the details about the Runloop class.
 90      */
 91     function Runloop(oc) {
 92         
 93         this._config = {
 94             'fps':                  DEFAULT_FPS, // if fps <= 0, only run 1st frame
 95             'useOptimizedRunloop':  false // overwrite fps to undefined
 96         };
 97 
 98         this._oc = oc;
 99 
100         this._shouldRun = false;
101         this._isLooping = false;
102 
103         this._lastFrame = null;
104         this._curFPS = null;
105 
106         this._hooks = [];
107         this._postHooks = []; // stack-ish: first-in-last-out
108 
109         this._requestFrameFunc = null;
110 
111         var gConfig = this._oc.getGlobalConfig();
112         OC.Util.sync(this._config, gConfig);
113 
114         if (this._config.fps > 0) {
115             if (this._config.useOptimizedRunloop) {
116                 this._config.fps = undefined;
117                 gConfig.fps = undefined;
118             }
119             this._requestFrameFunc = _requestLoopFrame(this._config.fps);
120         }
121     }
122 
123     // == Protected Methods ==
124 
125     Runloop.prototype._startLoop = function() {
126         if (this._isLooping) {
127             return;
128         }
129         
130         if (!!this._requestFrameFunc) {
131             this._shouldRun = true;
132         }
133         _loop(this);
134 
135     };
136 
137     Runloop.prototype._stopLoop = function() {
138         if (this._isLooping) {
139             this._shouldRun = false;
140         }
141     };
142 
143     // == private ==
144 
145     function _loop (rl) {
146 
147         (function run() {
148             _tick(rl);
149 
150             if (rl._shouldRun) {
151                 rl._requestFrameFunc(run);
152                 rl._isLooping = true;
153             } else {
154                 rl._isLooping = false;
155             }
156         })();
157     
158     }
159 
160     function _tick (rl) {
161         _calcCurFrame(rl);
162         _callMainHooks(rl);
163         _callPostHooks(rl);
164     }
165 
166     function _callMainHooks (rl) {
167         _callhooks(rl, rl._hooks);
168     }
169 
170     function _callPostHooks (rl) {
171         _callhooks(rl, rl._postHooks);
172     }
173 
174     function _callhooks (rl, hooks) {
175         var i, hook;
176         var count = hooks.length;
177 
178         for (i = 0; i < count; i++) {
179             hook = hooks[i];
180             hook(rl._oc);
181         }
182     }
183 
184     // fps: if omitted, let browser optimize
185     function _requestLoopFrame (fps) {
186 
187         var frameFunc = null;
188 
189         var st = window.setTimeout;
190         
191         var reqAnimFrame =  window.requestAnimationFrame       ||
192                             window.webkitRequestAnimationFrame || 
193                             window.mozRequestAnimationFrame    || 
194                             window.oRequestAnimationFrame      || 
195                             window.msRequestAnimationFrame     ||
196                             st;
197 
198         var func;
199 
200         if ( fps > 0 ) {
201             func = st;            
202         } else if ( fps === undefined ) {
203             func = reqAnimFrame;
204             fps = DEFAULT_FPS;
205         }
206 
207         if (func === st) {
208             frameFunc = function(callback) {
209                  func(callback, 1000 / fps);
210             };
211         } else if ( func !== undefined ) {
212             // func is the browser's requestAnimationFrame
213             frameFunc = function(callback) {
214                 func(callback);
215             };
216         }
217 
218         return frameFunc;
219     }
220 
221     function _calcCurFrame (rl) {
222         var dt, fps;
223         var curFrame = new Date().getTime();
224         dt = curFrame - rl._lastFrame;
225         if (~~dt) {
226             rl._curFPS = Math.ceil(1000 / dt);
227         }
228         rl._lastFrame = curFrame;
229     }
