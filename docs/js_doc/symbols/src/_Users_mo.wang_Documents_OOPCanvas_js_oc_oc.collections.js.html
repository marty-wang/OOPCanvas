1 //= require "oc.bootstrapper"
  2 //= require "oc.util"
  3 
  4 (function(OC) {
  5    
  6     /**
  7      * @namespace
  8      */
  9     OC.Collections = {};
 10 
 11     /**
 12      * @class Sortable Dictionary structure
 13      */
 14     OC.Collections.Dict = function Dict (sort) {
 15         this._dict = {};
 16         this._orders = [];
 17         this._sort = sort;
 18         if (!!this._sort) {
 19             var d = this;
 20             var s = d._sort;
 21             this._sortWrapper = function(k1, k2) {
 22                 var v1 = d._dict[k1];
 23                 var v2 = d._dict[k2];
 24                 return s(v1, v2);
 25             };
 26         }
 27     };
 28     
 29     /** 
 30      * @exports Dict as OOPCanvas.Collections.Dict
 31      */
 32     var Dict = OC.Collections.Dict;
 33     
 34     /**
 35      * return the value for the key.
 36      */
 37     Dict.prototype.getValue = function(key) {
 38         return this._dict[key]; 
 39     };
 40 
 41     /**
 42      * return if it is sorted dictionary.
 43      */
 44     Dict.prototype.isSorted = function() {
 45         return !!this._sort;
 46     };
 47 
 48     /**
 49      * return if it contains the key
 50      */
 51     Dict.prototype.contain = function (key) {
 52         return (key in this._dict);
 53     };
 54 
 55     /**
 56      * add the key-value pair to the dictionary
 57      */
 58     Dict.prototype.add = function(key, value) {
 59         if ( key in this._dict) {
 60             return;
 61         }
 62 
 63         this._dict[key] = value;
 64         if (!!this._sort) {
 65             this._orders.push(key);
 66             this._orders.sort(this._sortWrapper);
 67         }
 68     };
 69 
 70     /**
 71      * add an array of key-value pair to the dictionary
 72      * @param {Array[]} kvs each of the key-value pair is a 2-item array. kv[0]: key. kv[1]: value
 73      */
 74     Dict.prototype.addItems = function(kvs) {
 75         var i, kv, key, value;
 76         var count = kvs.length;
 77         
 78         for ( i = 0; i < count; i++ ) {
 79             kv = kvs[i];
 80             key = kv[0];
 81             value = kv[1];
 82 
 83             if ( key in this._dict ) {
 84                 continue;
 85             }
 86 
 87             this._dict[key] = value;
 88 
 89             if (!!this._sort) {
 90                 this._orders.push(key);
 91             }
 92         }
 93 
 94         if (!!this._sort) {
 95             this._orders.sort(this._sortWrapper);
 96         }
 97     };
 98 
 99     /**
100      * remove a key-value pair from the dictionary
101      */
102     Dict.prototype.remove = function(key) {
103         if ( key in this._dict ) {
104             delete this._dict[key];
105             if (!!this._sort) {
106                 OC.Util.arrayRemoveObject(this._orders, key);
107                 this._orders.sort(this._sortWrapper);
108             }
109         }    
110     };
111 
112     /**
113      * remove an array of keys from the dictionary
114      */
115     Dict.prototype.removeItems = function(keys) {
116         if ( typeof keys === "undefined" ) {
117             this.removeAll();
118         } else {
119             var i, key;
120             var count = keys.length;
121             for ( i = 0; i < count; i++ ) {
122                 key = keys[i];
123                 if ( key in this._dict ) {
124                     delete this._dict[key];
125                     if (!!this._sort) {
126                         OC.Util.arrayRemoveObject(this._orders, key);
127                     }
128                 }
129             }
130             this._orders.sort(this._sortWrapper);
131         }
132     };
133 
134     /**
135      * remove all key-value pairs from the dictionary
136      */
137     Dict.prototype.removeAll = function() {
138         this._dict = {};
139         this._orders.length = 0;
140     };
141 
142     /** iterate the dictionary
143      * @param {Function} callback The callback function will have the key and
144      * value as the two arguments
145      */
146     Dict.prototype.iterate = function(callback) {
147         var key, value;
148         if (!!this._sort) {
149             var i;
150             var num = this._orders.length;
151             for (i = 0; i < num; i++) {
152                 key = this._orders[i];
153                 value = this._dict[key];
154                 callback(key, value);
155             }
156         } else {
157             for ( key in this._dict ) {
158                 if (OOPCanvas.Util.hasOwnProperty(this._dict, key)) {
159                     value = this._dict[key];
160                     callback(key, value);
161                 }
162             }
163         }
164     };
