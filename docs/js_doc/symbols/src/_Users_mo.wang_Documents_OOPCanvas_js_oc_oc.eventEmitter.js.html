1 //= require "oc.bootstrapper"
  2 //= require "oc.util"
  3 
  4 (function (OC) {
  5 
  6     // ++ Augment OOPCanvas ++
  7    
  8     OC.EventEmitter = EventEmitter;
  9 
 10     /**
 11      * shorthand method to create Event Emitter instance.
 12      */
 13     OC.prototype.eventEmitter = function () {
 14         return new EventEmitter();
 15     };
 16 
 17     // == EventEmitter ==
 18 
 19     /**
 20      * Can be created by using shorthand method {@link OOPCanvas#eventEmitter}
 21      * @name EventEmitter
 22      * @exports EventEmitter as OOPCanvas.EventEmitter
 23      * @class The class deals with eventing.
 24      */
 25     function EventEmitter () {
 26         this._eventsMapper = {};
 27     }
 28 
 29     /**
 30     * @param { String } eventName
 31     * @param { Object|Function } subscriber Subscriber can be either object or
 32     * function. If it is an object, it must have an event handler function named
 33     * after "oneventName". For example, if the event is "created", the
 34     * corresponding event handler function should be named as "oncreated". If the
 35     * subscriber is a function, there is no naming striction.
 36     */
 37     EventEmitter.prototype.subscribe = function(eventName, subscriber) {
 38         if(typeof subscriber !== "object" && typeof subscriber !== "function") {
 39             throw "The subcriber needs to be either an object or a function";
 40         }
 41 
 42         var mapper = this._eventsMapper;
 43         if (!Object.prototype.hasOwnProperty.call(mapper, eventName)) {
 44             mapper[eventName] = [];
 45         }
 46 
 47         mapper[eventName].push(subscriber);
 48     };
 49 
 50     /**
 51      * @param {String} eventName
 52      * @param {Object|Function} [subscriber] if subscriber is left out, it will remove all the subscribers for that event.
 53      */
 54     EventEmitter.prototype.unsubscribe = function(eventName, subscriber) {
 55         var subscribers = this._eventsMapper[eventName];
 56 
 57         if (subscribers === undefined) {
 58             return;
 59         }
 60 
 61         if (subscriber === undefined) {
 62             subscribers.length = 0;
 63         } else {
 64             var index = OC.Util.arrayIndexOf(subscribers, subscriber);
 65             if (index > -1) {
 66                 subscribers.splice(index, 1);
 67             }
 68         }
 69 
 70         if(subscribers.length === 0) {
 71             delete this._eventsMapper[eventName];
 72         }
 73     };
 74 
 75     /**
 76      * trigger event
 77      */
 78     EventEmitter.prototype.fire = function(sender, eventName, eventArgs) {
 79         var subscribers = this._eventsMapper[eventName];
 80 
 81         if(subscribers === undefined) {
 82             return;
 83         }
 84 
 85         var num = subscribers.length;
 86         var i;
 87         var subscriber;
 88         var method;
 89 
 90         for (i = 0; i < num; i++) {
 91             subscriber = subscribers[i];
 92             if (typeof subscriber === "function") {
 93                 subscriber.call(null, sender, eventArgs);
 94             } else if (typeof subscriber === "object") {
 95                 method = "on" + eventName;
 96                 subscriber[method].call(subscriber, sender, eventArgs);
 97             }
 98         }
 99     };
100 
101     debug.info("eventEmitter module is installed.");
102 
103 })(OOPCanvas);
