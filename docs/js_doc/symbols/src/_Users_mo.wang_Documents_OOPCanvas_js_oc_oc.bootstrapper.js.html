 27 /**
 28  * Dependencies: <a href="http://benalman.com/projects/javascript-debug-console-log/">JavaScript Debug</a>, <a href="http://www.broofa.com/2008/09/javascript-uuid-function/">Math.uuid.js</a>, <a href="http://www.kevlindev.com/dom/path_parser/index.htm">Path Parser</a>
 29  * @class
 30  * @exports OC as OOPCanvas
 31  */
 32 var OOPCanvas = (function() {
 33     
 34     debug.setLevel(0);
 35     
 36     /**
 37      * @static
 38      */
 39     OOPCanvas.meta = {
 40         'version': '0.0.0',
 41         'author': 'Mo Wang',
 42         'repo': 'git@github.com:marty-wang/OOPCanvas.git',
 43         'license': 'MIT'
 44     };
 45 
 46     /**
 47      * @private
 48      * @constructor
 49      */
 50     function OOPCanvas () {}
 51 
 52     
 53     /**
 54      * @private
 55      */
 56     OOPCanvas._inits = [];
 57   
 58     /**
 59      * Plugins should be installed via this method
 60      * @param {String} namespace
 61      * @param {Function} plugin
 62      */
 63     OOPCanvas.installPlugin = function(namespace, plugin) {
 64         var fn = OOPCanvas.prototype;
 65 
 66         fn[namespace] = function() {
 67             if (!!plugin) {
 68                 return plugin.apply(this, arguments);
 69             }
 70         };
 71     };
 72 
 73     /**
 74      * @static
 75      */
 76     OOPCanvas.initialize = function(init) {
 77         OOPCanvas._inits.push(init);
 78     };
 79 
 80     
 81     OOPCanvas.prototype._initModules = function() {
 82         debug.info("+++ Start to Init Modules +++");
 83         var oc = this;
 84         var i, init;
 85         var inits = OOPCanvas._inits;
 86         var count = inits.length;
 87 
 88         for (i = 0; i < count; i++) {
 89             init = inits[i];
 90             init(oc);
 91         }
 92         debug.info("+++ Completed Modules Init +++");
 93     };
 94         
 95     return OOPCanvas;
 96 
 97 })();
 98 
 99 'OC' in window || (window.OC = OOPCanvas);
