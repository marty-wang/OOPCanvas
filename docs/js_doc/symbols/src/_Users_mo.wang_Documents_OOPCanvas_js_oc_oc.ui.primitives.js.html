 46     /**
 47      * @name Rectangle
 48      * @exports Rectangle as OOPCanvas.UIPrimitives.Rectangle
 49      * @class
 50      */
 51     function Rectangle (oc, left, top, width, height) {
 52         Rectangle.__super(this, 'constructor', oc);
 53 
 54         this._left = left;
 55         this._top = top;
 56         this._width = width;
 57         this._height = height;
 58         
 59         this._id = "Rect-" + this._id;
 60     }
 61 
 62     OC.Util.inherit(Rectangle, OC.UIElement);
 63             
 64     Rectangle.prototype._draw = function() {

 66         oc.drawRectangle(this._left, this._top, this._width, this._height, this._config);
 67     };
 68 
 69     Rectangle.prototype._hitTest = function(x, y) {
 70         var oc = this._oc;
 71         oc.drawRectangle(this._left, this._top, this._width, this._height);
 72         return this.testPointInPath(x, y);
 73     };
