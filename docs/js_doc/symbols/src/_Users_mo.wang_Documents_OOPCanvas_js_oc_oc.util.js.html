 37     /**
 38      * @param {Object} merging the object is merging
 39      * @param {Object} merged the object is being merged
 40      * @param {Boolean} [mutable="false"] wether or not the merging object will be
 41      * mutated
 42      * @param {Array} [excludes="[]"] the array containing the names of properties that should not be
 43      * merged
 44      * @returns the object of merging result
 45      */
 46     OC.Util.merge = function _merge (merging, merged/*, mutable, excludes*/) {
 47         var arg2 = arguments[2];
 48         var arg3 = arguments[3];
 49 
 50         // default values
 51         var mutable = false;
 52         var excludes = [];
 53 
 54         if (typeof arg2 === "boolean") {
 55             mutable = arg2;
 56             if (typeof arg3 === "object") {
 57                 excludes = arg3;
 58             }
 59         } else if (typeof arg2 === "object") {
 60             excludes = arg2;
 61         }
 62 
 63         var orig = {};
 64         if (!mutable) {
 65             orig = _merge(orig, merging, true);
 66         } else {
 67             orig = merging;
 68         }
 69 
 70         var prop;
 71         
 72         for (prop in merged) {
 73             if ( excludes.indexOf(prop) > -1 ) {
 74                 continue;
 75             }
 76             if (OC.Util.hasOwnProperty(merged, prop)) {
 77                 orig[prop] = merged[prop];
 78             }
 79         }
 80 
 81         return orig;
 82     };

 88         var k;
 89         for ( k in obj1 ) {
 90             if (OOPCanvas.Util.hasOwnProperty(obj1, k)) {
 91                 if ( k in obj2 ) {
 92                     obj1[k] = obj2[k];
 93                 } else {
 94                     obj2[k] = obj1[k];
 95                 }
 96             }
 97         }

109     /**
110      * Helper function that implements (pseudo)Classical inheritance inheritance.
111      * @see http://www.yuiblog.com/blog/2010/01/06/inheritance-patterns-in-yui-3/
112      * @param {Function} childClass
113      * @param {Function} parentClass
114      */
115     OC.Util.inherit = function (childClass, parentClass) {
116         var tempClass = function() {
117         };
118         tempClass.prototype = parentClass.prototype;
119         childClass.prototype = new tempClass();            
120         childClass.prototype.constructor = childClass;
121         childClass.__super = function(obj, methodName) {
122             var func = parentClass.prototype[methodName];
123             if ( !!func ) {
124                 var args = Array.prototype.slice.call(arguments, 2);
125                 return func.apply(obj, args);
126             }
127         };
128     };

132     /**
133      * @param {HTMLElement} element
134      * @returns the absolute position of the dom element in the HTML document
135      */
136     OC.Util.domElementPosition = function (obj){
137         var curleft = 0;
138         var curtop = 0;
139         if(obj.offsetParent) {
140             while(1) {
141                 curleft += obj.offsetLeft;
142                 curtop += obj.offsetTop;
143                 if(!obj.offsetParent) {
144                     break;
145                 }
146                 obj = obj.offsetParent;
147             }
148         } else {
149             if(obj.x) {
150                 curleft += obj.x;
151             }
152 
153             if(obj.y) {
154                 curtop += obj.y;
155             }
156         }
157         return {
158             'left': curleft,
159             'top': curtop
160         };
161     };

170     OC.Util.arrayIndexOf = function(array, obj) {
171         var idx = -1;
172         if (array.indexOf) {
173             idx = array.indexOf(obj);
174         } else {
175             var i, count;
176             for (i = 0, count = array.length; i < count; i++) {
177                 var o = array[i];
178                 if ( o === obj ) {
179                     idx = i;
180                     break;
181                 }
182             }
183         }
184 
185         return idx;
186     };
