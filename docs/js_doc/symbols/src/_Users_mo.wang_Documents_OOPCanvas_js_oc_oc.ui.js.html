1 //= require "oc.bootstrapper"
  2 //= require "oc.util"
  3 //= require "oc.drawing"
  4 //= require "oc.animation"
  5 //= require "oc.interaction"
  6 //= require "oc.eventEmitter"
  7 
  8 // UIElement is the base class of all visual elements on the canvas.
  9 // It is responsible for animation, hittest, event, update, render
 10 
 11 // Suppored Events: click, mouseover, mousemove, mouseout
 12 
 13 (function(OC) {
 14     
 15     // == Augment OOPCanvas ==
 16 
 17     OC.UIElement = UIElement;
 18 
 19     /**
 20      * @namespace
 21      */
 22     OC.UIControls = {};
 23     
 24     /**
 25      * @name UIElement
 26      * @exports UIElement as OOPCanvas.UIElement
 27      * @class The base class for all the UI Classes to extend
 28      */
 29     function UIElement(oc) {
 30         this._oc = oc;
 31 
 32         this._id = OC.Util.rand();
 33         this._zIndex = 0;
 34 
 35         this._left = null;
 36         this._top = null;
 37         this._centerX = null;
 38         this._centerY = null;
 39         this._config = null;
 40         
 41         this._isDirty = true;
 42         this._isHitTestVisible = true;
 43     
 44         this._eventEmitter = this._oc.eventEmitter();
 45         this._animator = null;
 46     }
 47 
 48     /**
 49      * @field
 50      */
 51     UIElement.Max_ZIndex = Number.MAX_VALUE;
 52     
 53     /**
 54      * @field
 55      */
 56     UIElement.Min_ZIndex = -UIElement.Max_ZIndex;
 57     
 58     /**
 59      * @returns the id of the UI Element
 60      */
 61     UIElement.prototype.getId = function() {
 62         return this._id;
 63     };
 64     
 65     UIElement.prototype.getZIndex = function() {
 66         return this._zIndex;
 67     };
 68 
 69     UIElement.prototype.setZIndex = function(index) {
 70         this._zIndex = index;
 71 
 72         this.invalidate();
 73     };
 74 
 75     UIElement.prototype.invalidate = function() {
 76         this._isDirty = true;
 77     };
 78 
 79     UIElement.prototype.isDirty = function() {
 80         return this._isDirty;
 81     };
 82 
 83     UIElement.prototype.setHitTestVisible = function(hitTestVisible) {
 84         this._isHitTestVisible = hitTestVisible;
 85     };
 86 
 87     UIElement.prototype.config = function(property, value) { 
 88         if ( !this._config ) {
 89             this._config = {};
 90         }
 91 
 92         if ( typeof property === "object" ) {
 93             var oc = this._oc;
 94             oc.updateConfig(this._config, property);
 95         } else {
 96             this._config[property] = value;
 97         }
 98 
 99         this.invalidate();
100     };
101 
102     UIElement.prototype.setPosition = function(left, top) {
103         this._left = left;
104         this._top = top;
105         
106         this.invalidate();
107     };
108 
109     UIElement.prototype.bind = function(eventName, callback) {
110         this._eventEmitter.subscribe(eventName, callback);
111     };
112 
113     UIElement.prototype.unbind = function(eventName, callback) {
114         this._eventEmitter.unsubscribe(eventName, callback);
115     };
116 
117     UIElement.prototype.fire = function(eventName, eventArgs) {
118         this._eventEmitter.fire(this, eventName, eventArgs);
119     };
120 
121     // shortcuts for binding event handlers
122     UIElement.prototype.click = function(callback) {
123         this.bind('click', callback); 
124     };
125 
126     UIElement.prototype.animate = function(props, duration, easingFunc) {
127         if ( !this._animator ) {
128             this._animator = this._oc.animator(this); 
129         }
130 
131         this._animator.start(props, duration, easingFunc);
132     };
133     
134     UIElement.prototype.update = function(currentTime) {
135         return this._update(currentTime);
136     };
137 
138     UIElement.prototype.draw = function() {
139         this._draw();
140         this._isDirty = false;
141     };
142 
143     UIElement.prototype.hitTest = function(x, y) {
144         if ( !this._isHitTestVisible ) {
145             return null;
146         }
147 
148         var oc  = this._oc;
149         var size = oc.getSize();
150         var w = size.width;
151         var h = size.height;
152         var ctx = oc._getHitTestContext();
153         ctx.clearRect(0, 0, w, h);
154 
155         // swap the rendering context with the hittest context
156         var tmp = oc.getContext;
157         oc.getContext = function() {
158             return ctx;
159         };
160 
161         var result = this._hitTest(x, y);
162         
163         oc.getContext = tmp;
164 
165         return result;        
166     };
167 
168     UIElement.prototype.testPointInPath = function(x, y) {
169         return this._oc._hitTest(this, x, y);
170     };
171 
172     // == Method to Override ==
173 
174     UIElement.prototype._update = function(currentTime) {
175         if (!!this._animator) {
176             this._isDirty |= this._animator.update(currentTime);
177         }
178 
179         return !!this._isDirty;
180     };
181 
182     // all the sub-classes must call this method
183     UIElement.prototype._draw = function() {};
184     
185     // sub-class needs to implement this method
186     // if it needs to participant to the hittest
187     UIElement.prototype._hitTest = function(x, y) {};
188 
189     // -- Event Handlers --
190     // Override the evenhanlder if subclass wants custom action
191     // before the event fires up
192 
193     UIElement.prototype._click = function() {
194         this.fire('click');
195     };
196 
197     UIElement.prototype._mouseover = function() {
198         this.fire('mouseover');
199     };
200 
201     UIElement.prototype._mousemove = function() {
202         this.fire('mousemove');
203     };
204 
205     UIElement.prototype._mouseout = function() {
206         this.fire('mouseout');
207     };
208 
209     UIElement.prototype._mousedown = function() {
210         this.fire('mousedown');
211     };
212 
213     UIElement.prototype._mouseup = function() {
214         this.fire('mouseup');
215     };
216 
217     // == End of Methods to Override ==
218     
219     debug.info("ui module UIElement submodule is installed.");
220 
221 })(OOPCanvas);
